オブジェクトの作り方

  • オブジェクトの設計図を作る
  • 設計図 → クラス
  • 1つの設計図（クラス）から多数のオブジェクトを生成
  • オブジェクトの1つ1つは異なる値をフィールドに保存
同じ種類のスマートフォン
色，メモリ容量，機種番号等異なる．
しかし，元の設計図は同じ

オブジェクトを作る

09_01

		
JitterBug bug; // オブジェクトを宣言
void setup() {
  size(480, 120);
  // オブジェクトを生成し，パラメータを渡す
  bug = new JitterBug(width/2, height/2, 20);
}
void draw() {
  bug.move();
  bug.display();
}
class JitterBug {
  float x;
  float y;
  int diameter;
  float speed = 2.5;
  JitterBug(float tempX, float tempY, int tempDiameter) {
    x = tempX;
    y = tempY;
    diameter = tempDiameter;
  }
  void move() {
    x += random(-speed, speed);
    y += random(-speed, speed);
  }
  void display() {
    ellipse(x, y, diameter, diameter);
  }
}

複数のオブジェクトを作る

09_02

		
JitterBug jit;
JitterBug bug;
void setup() {
  size(480, 120);
  jit = new JitterBug(width * 0.33, height/2, 50);
  bug = new JitterBug(width * 0.66, height/2, 10);
}
void draw() {
  jit.move();
  jit.display();
  bug.move();
  bug.display();
}
class JitterBug {
  float x;
  float y;
  int diameter;
  float speed = 2.5;
  JitterBug(float tempX, float tempY, int tempDiameter) {
    x = tempX;
    y = tempY;
    diameter = tempDiameter;
  }
  void move() {
    x += random(-speed, speed);
    y += random(-speed, speed);
  }
  void display() {
    ellipse(x, y, diameter, diameter);
  }
}
