オブジェクトとは？

  1. オブジェクトはクラスから生成される

  3. 入力に対して出力
  4. 入力に対して動作

オブジェクトを作る

09_01

		
JitterBug bug; // Declare object
void setup() {
  size(480, 120);
  // Create object and pass in parameters
  bug = new JitterBug(width/2, height/2, 20);
}
void draw() {
  bug.move();
  bug.display();
}
class JitterBug {
  float x;
  float y;
  int diameter;
  float speed = 2.5;
  JitterBug(float tempX, float tempY, int tempDiameter) {
    x = tempX;
    y = tempY;
    diameter = tempDiameter;
  }
  void move() {
    x += random(-speed, speed);
    y += random(-speed, speed);
  }
  void display() {
    ellipse(x, y, diameter, diameter);
  }
}

複数のオブジェクトを作る

09_02

		
JitterBug jit;
JitterBug bug;
void setup() {
  size(480, 120);
  jit = new JitterBug(width * 0.33, height/2, 50);
  bug = new JitterBug(width * 0.66, height/2, 10);
}
void draw() {
  jit.move();
  jit.display();
  bug.move();
  bug.display();
}
class JitterBug {
  float x;
  float y;
  int diameter;
  float speed = 2.5;
  JitterBug(float tempX, float tempY, int tempDiameter) {
    x = tempX;
    y = tempY;
    diameter = tempDiameter;
  }
  void move() {
    x += random(-speed, speed);
    y += random(-speed, speed);
  }
  void display() {
    ellipse(x, y, diameter, diameter);
  }
}
