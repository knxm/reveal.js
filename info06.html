}

  strokeWeight(weight);
  line(mouseX, mouseY, pmouseX, pmouseY);
}

float x;
float easing = 0.01;

void setup() {
  size(220, 120);
}

void draw() {
  float targetX = mouseX;
  x += (targetX - x) * easing;
  ellipse(x, 40, 12, 12);
  println(targetX + " : " + x);
}
